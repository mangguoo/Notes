1. 传入对象的key会隐式转换为字符串：

   ``` js
   var a=true;
   var obj={}
   obj[a]=2;
   
   console.log(obj.true); // 2
   
   var o={a:1};
   var o1={a:2};
   var obj={};
   obj[o]=10;
   console.log(obj[o1]); // 10
   
   var obj={a:1};
   var arr=["a"];
   console.log(obj[arr]); // 1
   ```

2. 使用forin循环时，Key会被转换为字符串，并且forin遍历数组时不会遍历空元素，遍历对象时不会遍历不可枚举属性：

   ``` js
   const arr = [1, 2, 3, , 4, undefined, 5]
   
   console.log(arr[3]) // undefined
   
   for (const key in arr) {
     console.log(key, typeof key)
   }
   // 0 string
   // 1 string
   // 2 string
   // 4 string
   // 5 string
   // 6 string
   
   console.log(3 in arr) // false
   console.log(5 in arr) // true
   // 只要元素不为空，in运算符则返回true
   ```

3. 对象属性上的this指向当前对象上下文环境中的this：

   ``` js
   var a = 10
   const obj = {
     a: 1,
     b: this.a
   }
   console.log(obj.b) // 10
   
   var a = 100
   const o = {
     a: 1,
     b: function () {
       const o1 = {
         a: 10,
         c: this.a // 这里的this指向对象o
       }
       console.log(o1.c)
     }
   }
   o.b() // 1
   ```

4. null用来在垃圾回收时。可以将不使用的引用类型回收掉：

   ``` js
   var o={a:1,b:2};
   o=null;
   ```
